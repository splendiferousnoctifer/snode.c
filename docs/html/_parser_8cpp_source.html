109  ssize_t Parser::readHeaderLine() {
110  ssize_t consumed = 0;
111  ssize_t ret = 0;
112 
113  do {
114  char ch = 0;
115  ret = socketContext->readFromPeer(&ch, 1);
116 
117  if (ret > 0) {
118  if (ch == '\r' || ch == '\n') {
119  consumed++;
120  if (ch == '\n') {
121  if (EOL) {
122  splitHeaderLine(line);
123  line.clear();
124  if (parserState != ParserState::ERROR) {
125  parserState = parseHeader();
126  }
127  EOL = false;
128  } else if (line.empty()) {
129  if (parserState != ParserState::ERROR) {
130  parserState = parseHeader();
131  }
132  } else {
133  EOL = true;
134  }
135  }
136  } else if (EOL) {
137  if (std::isblank(ch)) {
138  if ((hTTPCompliance & HTTPCompliance::RFC7230) == HTTPCompliance::RFC7230) {
139  parserState = parsingError(400, "Header Folding");
140  } else {
141  line += ch;
142  consumed++;
143  }
144  } else {
145  splitHeaderLine(line);
146  line.clear();
147  line += ch;
148  consumed++;
149  }
150  EOL = false;
151  } else {
152  line += ch;
153  consumed++;
154  }
155  } else if (ret < 0) {
156  consumed = ret;
157  }
158  } while (ret > 0 && parserState == ParserState::HEADER);
159 
160  return consumed;
161  }

163  void Parser::splitHeaderLine(const std::string& line) {
164  if (!line.empty()) {
165  std::string headerFieldName;
166  std::string headerFieldValue;
167  std::tie(headerFieldName, headerFieldValue) = httputils::str_split(line, ':');
168 
169  if (headerFieldName.empty()) {
170  parserState = parsingError(400, "Header-field empty");
171  } else if (std::isblank(headerFieldName.back()) || std::isblank(headerFieldName.front())) {
172  parserState = parsingError(400, "White space before or after header-field");
173  } else if (headerFieldValue.empty()) {
174  parserState = parsingError(400, "Header-value of field \"" + headerFieldName + "\" empty");
175  } else {
176  httputils::to_lower(headerFieldName);
177  httputils::str_trimm(headerFieldValue);
178 
179  if (headers.find(headerFieldName) == headers.end()) {
180  headers.insert({headerFieldName, headerFieldValue});
181  } else {
182  headers[headerFieldName] += "," + headerFieldValue;
183  }
184  }
185  } else {
186  parserState = parsingError(400, "Header-line empty");
187  }
188  }

190  ssize_t Parser::readContent() {
191  if (contentRead == 0) {
192  content = new char[contentLength];
193  }
194 
195  std::size_t contentJunkLenLeft =
196  (contentLength - contentRead < MAX_CONTENT_JUNK_LEN) ? contentLength - contentRead : MAX_CONTENT_JUNK_LEN;
197 
198  ssize_t ret = socketContext->readFromPeer(contentJunk, contentJunkLenLeft);
199 
200  if (ret > 0) {
201  std::size_t contentJunkLen = static_cast<std::size_t>(ret);
202  if (contentRead + contentJunkLen <= contentLength) {
203  memcpy(content + contentRead, contentJunk, contentJunkLen); // NOLINT(clang-analyzer-core.NonNullParamChecker)
204 
205  contentRead += contentJunkLen;
206  if (contentRead == contentLength) {
207  parserState = parseContent(content, contentLength);
208 
209  delete[] content;
210  content = nullptr;
211  contentRead = 0;
212  }
213  } else {
214  parserState = parsingError(400, "Content to long");
215 
216  if (content != nullptr) {
217  delete[] content;
218  content = nullptr;
219  }
220  }
221  }
222 
223  return ret;
224  }
