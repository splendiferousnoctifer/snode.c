42  const std::function<void(void)>& onTimeout,
43  const std::function<void(int err)>& onError)
44  : ReadEventReceiver(TLSHANDSHAKE_TIMEOUT)
45  , WriteEventReceiver(TLSHANDSHAKE_TIMEOUT)

54  int sslErr = SSL_get_error(ssl, ret);
55 
56  ReadEventReceiver::enable(fd);
57  WriteEventReceiver::enable(fd);
58  ReadEventReceiver::suspend();
59  WriteEventReceiver::suspend();
60 
61  switch (sslErr) {
62  case SSL_ERROR_WANT_READ:
63  ReadEventReceiver::resume();
64  break;
65  case SSL_ERROR_WANT_WRITE:
66  WriteEventReceiver::resume();
67  break;
68  case SSL_ERROR_NONE:
69  ReadEventReceiver::disable();
70  WriteEventReceiver::disable();
71  onSuccess();
72  break;
73  default:
74  ReadEventReceiver::disable();
75  WriteEventReceiver::disable();
