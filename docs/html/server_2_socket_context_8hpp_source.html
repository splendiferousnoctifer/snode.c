144  template <typename Request, typename Response>
145  void SocketContext<Request, Response>::requestParsed() {
146  if (!requestInProgress) {
147  RequestContext& requestContext = requestContexts.front();
148 
149  requestInProgress = true;
150  if (requestContext.status == 0) {
151  if (((requestContext.request.connectionState == ConnectionState::Close) ||
152  (requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
153  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
154  requestContext.request.connectionState != ConnectionState::Keep) ||
155  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
156  requestContext.request.connectionState == ConnectionState::Close))) {
157  requestContext.response.set("Connection", "close");
158  } else {
159  requestContext.response.set("Connection", "keep-alive");
160  }
161 
162  onRequestReady(requestContext.request, requestContext.response);
163  } else {
164  requestContext.response.status(requestContext.status).send(requestContext.reason);
165  terminateConnection();
166  }
167  }
168  }

170  template <typename Request, typename Response>
171  void SocketContext<Request, Response>::sendToPeerCompleted() {
172  RequestContext& requestContext = requestContexts.front();
173 
174  // if 0.9 => terminate
175  // if 1.0 && (request != Keep || contentLength = -1) => terminate
176  // if 1.1 && (request == Close || contentLength = -1) => terminate
177  // if (request == Close) => terminate
178 
179  if ((requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
180  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
181  requestContext.request.connectionState != ConnectionState::Keep) ||
182  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
183  requestContext.request.connectionState == ConnectionState::Close) ||
184  (requestContext.response.connectionState == ConnectionState::Close)) {
185  terminateConnection();
186  } else {
187  reset();
188 
189  requestContexts.pop_front();
190 
191  if (!requestContexts.empty() && requestContexts.front().ready) {
192  RequestContext& requestContext = requestContexts.front();
193 
194  requestInProgress = true;
195  if (requestContext.status == 0) {
196  onRequestReady(requestContext.request, requestContext.response);
197  } else {
198  requestContext.response.status(requestContext.status).send(requestContext.reason);
199  terminateConnection();
200  }
201  }
