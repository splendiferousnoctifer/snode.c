75  static bool matchFunction(const std::string& cpath, const std::string& reqpath) {
76  std::vector<std::string> explodedString = explode(cpath, '/');
77 
78  for (std::vector<std::string>::size_type i = 0; i < explodedString.size(); i++) {
79  if (explodedString[i].front() == ':') {
80  std::smatch smatch = matchResult(explodedString[i]);
81  std::string regex = "(.*)";
82 
83  if (smatch.size() > 1) {
84  if (smatch[1] != "") {
85  regex = smatch[1];
86  }
87  }
88 
89  explodedString[i] = regex;
90  }
91  }
92 
93  std::string regexPath = path_concat(explodedString);
94 
95  return std::regex_match(reqpath, std::regex(regexPath));
96  }

98  static void setParams(const std::string& cpath, Request& req) {
99  std::vector<std::string> explodedString = explode(cpath, '/');
100  std::vector<std::string> explodedReqString = explode(req.url, '/');
101 
102  for (std::vector<std::string>::size_type i = 0; i < explodedString.size() && i < explodedReqString.size(); i++) {
103  if (explodedString[i].front() == ':') {
104  std::smatch smatch = matchResult(explodedString[i]);
105  std::string regex = "(.*)";
106 
107  if (smatch.size() > 1) {
108  if (smatch[1] != "") {
109  regex = smatch[1];
110  }
111  }
112 
113  if (std::regex_match(explodedReqString[i], std::regex(regex))) {
114  std::string attributeName = smatch[0];
115  attributeName.erase(0, 1);
116  attributeName.erase((attributeName.length() - smatch[1].length()), smatch[1].length());
117 
118  req.params[attributeName] = explodedReqString[i];
119  }
120  }
121  }
122  }

129  static inline std::string path_concat(const std::string& first, const std::string& second) {
130  std::string result;
131 
132  if (first.back() == '/' && second.front() == '/') {
133  result = first + second.substr(1);
134  } else if (first.back() != '/' && second.front() != '/') {
135  result = first + '/' + second;
136  } else {
137  result = first + second;
138  }
139 
140  if (result.length() > 1 && result.back() == '/') {
141  result.pop_back();
142  }
143 
144  return result;
145  }
