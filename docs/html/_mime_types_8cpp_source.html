223  MimeTypes::MimeTypes() {
224 #ifdef HAS_LIBMAGIC
225  MimeTypes::magic = magic_open(MAGIC_MIME);
226 
227  if (magic_load(magic, nullptr) != 0) {
228  LOG(WARNING) << "cannot load magic database - " + std::string(magic_error(magic));
229  magic_close(magic);
230  magic = nullptr;
231  }
232 #endif
233  }

241  std::string MimeTypes::contentType(const std::string& file) {
242  std::map<std::string, std::string>::iterator it = mimeType.find(std::filesystem::path(file).extension());
243 
244  std::string type;
245 
246  if (it != mimeType.end()) {
247  type = it->second;
248  }
249 #ifdef HAS_LIBMAGIC
250  else if (magic != nullptr) {
251  type = magic_file(MimeTypes::magic, file.c_str());
252  }
253 #endif
254 
255  return type;
256  }
