119  void Request::enqueue(const char* junk, std::size_t junkLen) {
120  if (!headersSent && !sendHeaderInProgress) {
121  sendHeaderInProgress = true;
122  sendHeader();
123  sendHeaderInProgress = false;
124  headersSent = true;
125  }
126 
127  socketContext->sendToPeer(junk, junkLen);
128 
129  if (headersSent) {
130  contentSent += junkLen;
131  if (contentSent == contentLength) {
132  socketContext->sendToPeerCompleted();
133  } else if (contentSent > contentLength) {
134  socketContext->terminateConnection();
135  }
136  }
137  }

143  void Request::sendHeader() {
144  std::string httpVersion = "HTTP/" + std::to_string(httpMajor) + "." + std::to_string(httpMinor);
145 
146  std::string queryString = "";
147  if (queries.size() > 0) {
148  queryString += "?";
149  for (auto& [key, value] : queries) {
150  queryString += httputils::url_encode(key) + "=" + httputils::url_encode(value) + "&";
151  }
152  queryString.pop_back();
153  }
154 
155  enqueue(method + " " + url + queryString + " " + httpVersion + "\r\n");
156 
157  enqueue("Host: " + host + "\r\n");
158  enqueue("Date: " + httputils::to_http_date() + "\r\n");
159 
160  headers.insert({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
161 
162  for (const auto& [field, value] : headers) {
163  enqueue(field + ":" + value + "\r\n");
164  }
165 
166  if (contentLength != 0) {
167  enqueue("Content-Length: " + std::to_string(contentLength) + "\r\n");
168  }
169  /*
170  if (connectionState == ConnectionState::Keep) {
171  enqueue("Connection: keep-alive\r\n");
172  } else if (connectionState == ConnectionState::Close) {
173  enqueue("Connection: close\r\n");
174  }
175  */
176  for (const auto& [name, value] : cookies) {
177  enqueue("Cookie:" + name + "=" + value + "\r\n");
178  }
179 
180  enqueue("\r\n");
181 
182  if (headers.find("Content-Length") != headers.end()) {
183  contentLength = std::stoul(headers.find("Content-Length")->second);
184  } else {
185  contentLength = 0;
186  }
187  }

205  void Request::upgrade(const std::string& url, const std::string& protocol) {
206  this->url = url;
207 
208  set("Connection", "Upgrade", true);
209 
210  set("Upgrade", protocol);
211 
212  // load upgrade context
213  // let upgrade context fill the request-header fields
214  // send the request to the server
215  // the response-code needs to check the response from the server and upgrade the context in case the server says OK
216 
217  web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
218  web::http::client::SocketContextUpgradeFactorySelector::instance()->select(protocol, *this);
219 
220  if (socketContextUpgradeFactory != nullptr) {
221  start();
222  } else {
223  socketContext->terminateConnection();
224  }
225  }

227  void Request::upgrade(Response& response) {
228  if (httputils::ci_contains(response.header("connection"), "Upgrade")) {
229  web::http::client::SocketContextUpgradeFactory* socketContextUpgradeFactory =
230  web::http::client::SocketContextUpgradeFactorySelector::instance()->select(*this, response);
231 
232  if (socketContextUpgradeFactory != nullptr) {
233  if (socketContext->switchSocketContext(socketContextUpgradeFactory) == nullptr) {
234  socketContext->terminateConnection();
235  }
236  } else {
237  VLOG(0) << "SocketContextUpgradeFactory not existing";
238  socketContext->terminateConnection();
239  }
240 
241  } else {
242  VLOG(0) << "Response did not contain upgrade";
243  socketContext->terminateConnection();
244  }
245  }
