43  InetAddress::InetAddress(const std::string& ipOrHostname, uint16_t port) {
44  struct addrinfo hints {};
45  struct addrinfo* res;
46  struct addrinfo* resalloc;
47 
48  memset(&hints, 0, sizeof(hints));
49  memset(&sockAddr, 0, sizeof(sockAddr));
50 
51  /* We only care about IPV6 results */
52  hints.ai_family = AF_INET6;
53  hints.ai_socktype = 0;
54  hints.ai_flags = AI_ADDRCONFIG | AI_V4MAPPED;
55 
56  int err = net::system::getaddrinfo(ipOrHostname.c_str(), nullptr, &hints, &res);
57 
58  if (err != 0) {
59  throw bad_hostname(ipOrHostname);
60  }
61 
62  resalloc = res;
63 
64  while (res) {
65  /* Check to make sure we have a valid AF_INET6 address */
66  if (res->ai_family == AF_INET6) {
67  /* Use memcpy since we're going to free the res variable later */
68  memcpy(&sockAddr, res->ai_addr, res->ai_addrlen);
69 
70  /* Here we convert the port to network byte order */
71  sockAddr.sin6_port = htons(port);
72  sockAddr.sin6_family = AF_INET6;
73  break;
74  }
75 
76  res = res->ai_next;
77  }
78 
79  net::system::freeaddrinfo(resalloc);
80  }
