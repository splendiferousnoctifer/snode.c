71  enum Parser::ParserState RequestParser::parseStartLine(const std::string& line) {
72  enum Parser::ParserState parserState = Parser::ParserState::HEADER;
73 
74  if (!line.empty()) {
75  std::string remaining;
76 
77  std::tie(method, remaining) = httputils::str_split(line, ' ');
78  std::tie(url, httpVersion) = httputils::str_split(remaining, ' ');
79 
80  std::string queriesLine;
81  std::tie(std::ignore, queriesLine) = httputils::str_split(url, '?');
82 
83  if (!methodSupported(method)) {
84  parserState = parsingError(400, "Bad request method");
85  } else if (url.empty() || url.front() != '/') {
86  parserState = parsingError(400, "Malformed request");
87  } else {
88  std::smatch httpVersionMatch;
89  if (!std::regex_match(httpVersion, httpVersionMatch, httpVersionRegex)) {
90  parserState = parsingError(400, "Wrong protocol-version");
91  } else {
92  httpMajor = std::stoi(httpVersionMatch.str(1));
93  httpMinor = std::stoi(httpVersionMatch.str(2));
94 
95  while (!queriesLine.empty()) {
96  std::string query;
97 
98  std::tie(query, queriesLine) = httputils::str_split(queriesLine, '&');
99  queries.insert(httputils::str_split(query, '='));
100  }
101 
102  onRequest(method, url, httpVersion, httpMajor, httpMinor, queries);
103  }
104  }
105  } else {
106  parserState = parsingError(400, "Request-line empty");
107  }
108 
109  return parserState;
110  }

112  enum Parser::ParserState RequestParser::parseHeader() {
113  for (auto& [headerFieldName, headerFieldValue] : Parser::headers) {
114  if (headerFieldName != "cookie") {
115  if (headerFieldName == "content-length") {
116  Parser::contentLength = std::stoul(headerFieldValue);
117  }
118  } else {
119  std::string cookiesLine = headerFieldValue;
120 
121  while (!cookiesLine.empty()) {
122  std::string cookieLine;
123  std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
124 
125  while (!cookieLine.empty()) {
126  std::string cookie;
127  std::tie(cookie, cookieLine) = httputils::str_split(cookieLine, ';');
128 
129  std::string cookieName;
130  std::string cookieValue;
131  std::tie(cookieName, cookieValue) = httputils::str_split(cookie, '=');
132 
133  httputils::str_trimm(cookieName);
134  httputils::str_trimm(cookieValue);
135 
136  cookies.insert({cookieName, cookieValue});
137  }
138  }
139  }
140  }
141 
142  Parser::headers.erase("cookie");
143 
144  onHeader(Parser::headers, cookies);
145 
146  enum Parser::ParserState parserState = Parser::ParserState::BODY;
147  if (contentLength == 0) {
148  parsingFinished();
149  parserState = ParserState::BEGIN;
150  }
151 
152  return parserState;
153  }
