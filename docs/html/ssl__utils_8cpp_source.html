40  static int password_callback(char* buf, int size, int, void* u) {
41  strncpy(buf, static_cast<char*>(u), size);
42  buf[size - 1] = '\0';
43 
44  memset(u, 0, ::strlen(static_cast<char*>(u))); // garble password
45  free(u);
46 
47  return ::strlen(buf);

54  unsigned long ssl_init_ctx(SSL_CTX* ctx, const std::map<std::string, std::any>& options, bool server) {
55  unsigned long sslErr = 0;
56 
57  std::string certChain;
58  std::string keyPEM;
59  std::string password;
60  std::string caFile;
61  std::string caDir;
62  bool useDefaultCADir = false;
63 
64  for (auto& [name, value] : options) {
65  if (name == "certChain") {
66  certChain = std::any_cast<const char*>(value);
67  } else if (name == "keyPEM") {
68  keyPEM = std::any_cast<const char*>(value);
69  } else if (name == "password") {
70  password = std::any_cast<const char*>(value);
71  } else if (name == "caFile") {
72  caFile = std::any_cast<const char*>(value);
73  } else if (name == "caDir") {
74  caDir = std::any_cast<const char*>(value);
75  } else if (name == "useDefaultCADir") {
76  useDefaultCADir = std::any_cast<bool>(value);
77  }
78  }
79 
80  if (ctx != nullptr) {
81  if (!caFile.empty() || !caDir.empty()) {
82  if (!SSL_CTX_load_verify_locations(
83  ctx, !caFile.empty() ? caFile.c_str() : nullptr, !caDir.empty() ? caDir.c_str() : nullptr)) {
84  sslErr = ERR_peek_error();
85  }
86  }
87  if (sslErr == SSL_ERROR_NONE && useDefaultCADir) {
88  if (!SSL_CTX_set_default_verify_paths(ctx)) {
89  sslErr = ERR_peek_error();
90  }
91  }
92  if (server && sslErr == 0 && (useDefaultCADir || !caFile.empty() || !caDir.empty())) {
93  SSL_CTX_set_verify_depth(ctx, 5);
94  SSL_CTX_set_verify(ctx, SSL_VERIFY_FLAGS, verify_callback);
95  }
96  if (sslErr == SSL_ERROR_NONE) {
97  if (!certChain.empty()) {
98  if (SSL_CTX_use_certificate_chain_file(ctx, certChain.c_str()) <= 0) {
99  sslErr = ERR_peek_error();
100  } else if (!keyPEM.empty()) {
101  if (!password.empty()) {
102  SSL_CTX_set_default_passwd_cb(ctx, password_callback);
103  SSL_CTX_set_default_passwd_cb_userdata(ctx, ::strdup(password.c_str()));
104  }
105  if (SSL_CTX_use_PrivateKey_file(ctx, keyPEM.c_str(), SSL_FILETYPE_PEM) <= 0) {
106  sslErr = ERR_peek_error();
107  } else if (!SSL_CTX_check_private_key(ctx)) {
108  sslErr = ERR_peek_error();
109  }
110  }
111  }
112  }
113  } else {
114  sslErr = ERR_peek_error();
115  }
116 
117  return sslErr;
118  };

120  void ssl_log(const std::string& message, int sslErr) {
121  switch (sslErr) {
122  case SSL_ERROR_NONE:
123  case SSL_ERROR_ZERO_RETURN:
124  ssl_log_info(message);
125  break;
126  case SSL_ERROR_SYSCALL:
127  if (errno != 0) {
128  ssl_log_warning(message);
129  } else {
130  ssl_log_info(message + "- syscall error");
131  }
132  break;
133  default:
134  ssl_log_warning(message);
135  break;
136  };

186 /*
187  * Catch a too long certificate chain. The depth limit set using
188  * SSL_CTX_set_verify_depth() is by purpose set to "limit+1" so
189  * that whenever the "depth>verify_depth" condition is met, we
190  * have violated the limit and want to log this error condition.
191  * We must do it here, because the CHAIN_TOO_LONG error would not
192  * be found explicitly; only errors introduced by cutting off the
193  * additional certificates would be logged.
194  */
195 if (depth > SSL_get_verify_depth(ssl)) {
196  preverify_ok = 0;

199 }

203  LOG(ERROR) << "|-- message: " << X509_verify_cert_error_string(err);
204  LOG(ERROR) << "|-- depth: " << depth;
205  LOG(ERROR) << "|-- subject name: " << buf;
206 } else if (true) {
