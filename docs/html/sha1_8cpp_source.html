31 #define SHA1_R0(v, w, x, y, z, i) \
32  z += ((w & (x ^ y)) ^ y) + block[i] + 0x5a827999 + SHA1_ROL(v, 5); \
33  w = SHA1_ROL(w, 30);
34 #define SHA1_R1(v, w, x, y, z, i) \

73 std::string SHA1::final() {
74  /* Total number of hashed bits */
75  uint64_t total_bits = (transforms * BLOCK_BYTES + buffer.size()) * 8;
76 
77  /* Padding */
78  buffer += static_cast<std::string::value_type>(0x80);
79  std::size_t orig_size = buffer.size();
80  while (buffer.size() < BLOCK_BYTES) {
81  buffer += (char) 0x00;
82  }
83 
84  uint32_t block[BLOCK_INTS];
85  buffer_to_block(buffer, block);
86 
87  if (orig_size > BLOCK_BYTES - 8) {
88  transform(block);
89  for (unsigned int i = 0; i < BLOCK_INTS - 2; i++) {
90  block[i] = 0;
91  }
92  }
93 
94  /* Append total_bits, split this uint64 into two uint32 */
95  block[BLOCK_INTS - 1] = static_cast<uint32_t>(total_bits & 0x00000000FFFFFFFF);
96  block[BLOCK_INTS - 2] = static_cast<uint32_t>((total_bits >> 32) & 0x00000000FFFFFFFF);
97  transform(block);
98 
99  /* Hex std::string */
100  std::ostringstream result;
101  for (unsigned int i = 0; i < DIGEST_INTS; i++) {
102  result << std::hex << std::setfill('0') << std::setw(8);
103  result << (digest[i] & 0xffffffff);
104  }
105 
106  /* Reset for next run */
107  reset();
108 
109  return result.str();
110 }
