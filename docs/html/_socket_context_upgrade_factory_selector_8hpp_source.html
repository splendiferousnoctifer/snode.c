67  SocketContextUpgradeFactory*
68  SocketContextUpgradeFactorySelector<SocketContextUpgradeFactory>::load(const std::string& upgradeContextName,
69  typename SocketContextUpgradeFactory::Role role) {
70  SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
71 
72  for (const std::string& searchPath : searchPaths) {
73  void* handle =
74  net::DynamicLoader::dlOpen((searchPath + "/libsnodec-" + upgradeContextName + ".so").c_str(), RTLD_LAZY | RTLD_GLOBAL);
75 
76  if (handle != nullptr) {
77  SocketContextUpgradeFactory* (*getSocketContextUpgradeFactory)() =
78  net::DynamicLoader::dlSym<SocketContextUpgradeFactory* (*) ()>(
79  handle,
80  upgradeContextName + (role == SocketContextUpgradeFactory::Role::SERVER ? "Server" : "Client") +
81  "ContextUpgradeFactory");
82 
83  if (getSocketContextUpgradeFactory != nullptr) {
84  socketContextUpgradeFactory = getSocketContextUpgradeFactory();
85 
86  if (socketContextUpgradeFactory != nullptr) {
87  if (add(socketContextUpgradeFactory, handle)) {
88  VLOG(0) << "SocketContextUpgradeFactory created successfull: " << socketContextUpgradeFactory->name();
89  } else {
90  VLOG(0) << "UpgradeSocketContext already existing. Not using: " << socketContextUpgradeFactory->name();
91  socketContextUpgradeFactory->destroy();
92  socketContextUpgradeFactory = nullptr;
93  net::DynamicLoader::dlCloseDelayed(handle);
94  }
95  break;
96  } else {
97  net::DynamicLoader::dlCloseDelayed(handle);
98  VLOG(0) << "SocketContextUpgradeFactory not created: " << upgradeContextName;
99  }
100  } else {
101  net::DynamicLoader::dlCloseDelayed(handle);
102  VLOG(0) << "Not a Plugin \"" << upgradeContextName;
103  }
104  } else {
105  VLOG(0) << "Error dlopen: " << net::DynamicLoader::dlError();
106  }
107  }
108 
109  return socketContextUpgradeFactory;
110  }

113  SocketContextUpgradeFactory*
114  SocketContextUpgradeFactorySelector<SocketContextUpgradeFactory>::select(const std::string& upgradeContextName) {
115  SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
116 
117  if (!upgradeContextName.empty()) {
118  if (socketContextUpgradePlugins.contains(upgradeContextName)) {
119  socketContextUpgradeFactory = socketContextUpgradePlugins[upgradeContextName].socketContextUpgradeFactory;
120  } else if (linkedSocketContextUpgradePlugins.contains(upgradeContextName)) {
121  socketContextUpgradeFactory = linkedSocketContextUpgradePlugins[upgradeContextName]();
122  add(socketContextUpgradeFactory);
123  } else if (!onlyLinked) {
124  socketContextUpgradeFactory = load(upgradeContextName);
125  }
126  }
127 
128  return socketContextUpgradeFactory;
